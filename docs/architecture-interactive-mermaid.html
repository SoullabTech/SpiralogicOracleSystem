// Factory Pattern Implementation
import type { AgentState, AgentMemory, AgentPersonality } from './types';

/**
 * Creates a complete default AgentState
 * Single source of truth for all agent initialization
 */
export function createDefaultAgentState(
  userId: string,
  name: string = 'Your Oracle'
): AgentState {
  const timeOfDay = getCurrentTimeOfDay();

  return {
    id: `oracle-${userId}-${Date.now()}`,
    name,
    personality: createDefaultPersonality(),
    memory: createDefaultMemory(userId),
    currentContext: {
      timeOfDay,
      conversationDepth: 0,
      conversationState: 'casual',
      emotionalLoad: 0,
      realityLayers: {
        physical: 'entering sacred space',
        emotional: 'curiosity and openness',
        mental: 'questioning and seeking',
        spiritual: 'soul stirring'
      }
    },
    evolutionStage: 1,
    realityAwareness: createDefaultRealityAwareness()
  };
}
